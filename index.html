F1: Command Line Arguments & Switch

Basics / Logic 
public class F1a {
    public static void main(String[] args) {
        System.out.println("Message : " + args[0]);
    }
}
public class F1b {
    public static void main(String[] args) {

        double num1 = Double.parseDouble(args[0]);
        char op = args[1].charAt(0);
        double num2 = Double.parseDouble(args[2]);
        double res;

        switch (op) {
            case '+':
                res = num1 + num2;
                break;

            case '-':
                res = num1 - num2;
                break;

            case 'm':
                res = num1 * num2;
                break;

            case '/':
                if (num2 == 0) {
                    System.out.println("Division by zero not possible");
                    return;
                }
                res = num1 / num2;
                break;

            default:
                System.out.println("Invalid operator");
                return;
        }

        System.out.println("Result: " + res);
    }
}

F2: Scanner & Matrix Operations

Arrays / I/O 
import java.util.Scanner;

public class F2 {
    public static void main(String[] args) {
        if (args.length != 1) {
            System.out.println("Usage: java CommandLineInput <n>");
            return;
        }
        int n = Integer.parseInt(args[0]);
        Scanner in = new Scanner(System.in);
        int[][] mat1 = new int[n][n];  
        int[][] mat2 = new int[n][n];  
        int[][] Result = new int[n][n];  
        
        
        System.out.println("Enter matrix 1: ");
        for(int i = 0; i < n; i++){
            for (int j = 0; j < n; j++) {
                mat1[i][j] = in.nextInt();
            }
        }
        System.out.println("Enter matrix 2: ");
        for(int i = 0; i < n; i++){
            for (int j = 0; j < n; j++) {
                mat2[i][j] = in.nextInt();
            }
        }
        System.out.println();
        // sum 
        for(int i = 0; i < n; i++){
            for (int j = 0; j < n; j++) {
                Result[i][j] = mat1[i][j] + mat2[i][j];
                System.out.print(Result[i][j] +" ");
            }
            System.out.println();
        }

    }
}

F4: Method Overloading

Polymorphism / Math 
import java.util.Arrays;

class Point{
    int x;
    int y;
    Point(){
        x = 0;
        y = 0;
    }

    Point(int x, int y) {
        this.x = x;
        this.y = y;
    }
    public void setXY(int x, int y){
        this.x = x;
        this.y = y;
    }
    public int[] getXY(){
        return new int[]{x,y};
    }
    public String ToString(){
        return "(" + x + "," + y + ")";
    }
    public double distance(int x, int y){
        return Math.sqrt(Math.pow(this.x - x, 2) + Math.pow(this.y - y, 2));
    }
    public double distance(Point A){
        return distance(A.x, A.y);
    }
    public double distance(){
        return Math.sqrt(Math.pow(x, 2)+Math.pow(y, 2));
    }
}


public class F4 {
    public static void main(String[] args) {
        Point p1 = new Point(1,2);
        System.out.println(Arrays.toString(p1.getXY()));
        p1.ToString();
        Point p2 = new Point();
        p2.setXY(5, 5);
        System.out.println(Arrays.toString(p2.getXY()));
        p2.ToString();

        System.out.println(p1.distance(5,5));
        System.out.println(p1.distance(p2));
        System.out.println(p1.distance());
    }
}

class MyShape{
    void draw(){
        System.out.println("Drawing The Generic Shape");

        System.out.println("Erasing The Shape");
    }
}

    void draw(){
        System.out.println("Drawing The Square");
    }

        System.out.println("Erasing The Square");
    }
}
class Circle extends MyShape{
    void draw(){
        System.out.println("Drawing The Circle");
    }
    void erase(){
        System.out.println("Erasing The Circle");
    }
}
class Triangle extends MyShape{
    void draw(){
        System.out.println("Drawing The Triangle");
    }
    void erase(){
        System.out.println("Erasing The Triangle");
    }
}

F6: Interfaces

Abstraction 
interface Resizable{
    void resizeWidth(int width);
    void resizeHeight(int height);
}

class Rectangle implements Resizable{
    int height;
    int width;

    Rectangle(int height,int width) {
        this.height = height;
        this.width = width;
    }
    public void resizeHeight(int newHeight){
        this.height = newHeight;
    }
    public void resizeWidth(int newWidth){
        this.width = newWidth;
    }
    public void display(){
        System.out.println("Width: " + width + " " + "Height: " + height);
    }

}

public class F6 {
    public static void main(String[] args) {
        Rectangle r = new Rectangle(10, 20);
        r.display();
        r.resizeHeight(100);
        r.resizeWidth(200);
        r.display();
    }
}

F7: Custom Exception Handling

Try/Catch/Finally 
class DivisionByZero extends Exception{
    DivisionByZero(String msg) {
        super(msg);
    }
}

public class F7 {
    public static void main(String[] args) {
        try {
            int a = 100;
            int b = 0;
            if(b == 0) throw new DivisionByZero("Cannot divide by zero");

            System.out.println(a/b);
        }
        catch (DivisionByZero e){
            System.out.println(e.getMessage());
        }
        finally {
            System.out.println("Execution completed");
        }
    }
}
